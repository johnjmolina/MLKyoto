Define latex commands here ... $\newcommand{\bm}[1]{\mathbf{#1}} \newcommand{\df}[1]{\textrm{d}{#1}} \newcommand{\vdf}[1]{\df{\bm{#1}}}$

$$\begin{align*} P(\bm{X}\lvert I) + P(\overline{\bm{X}}\lvert I) &= 1 &\textrm{(Sum Rule)}\label{e:sum_rule}\\ P(\bm{X},\bm{Y}\lvert I) &= P(\bm{X}\lvert \bm{Y}, I)\times P(\bm{Y}\lvert I) &\textrm{(Product Rule)}\notag\\ &=P(\bm{Y}\lvert\bm{X}, I)\times P(\bm{X}\lvert I) \label{e:product_rule} \\ \end{align*}$$$$\begin{align*} \overbrace{P(\bm{X} \lvert \bm{Y}, I)}^{\small{\mathrm{posterior}}} &= \frac{\overbrace{P(\bm{Y}\lvert \bm{X},I)}^{\small{\mathrm{likelihood}}} \times\overbrace{P(\bm{X}\lvert I)}^{\small{\mathrm{prior}}}}{\underbrace{P(\bm{Y}\lvert I)}_{\small{\mathrm{evidence}}}}&\textrm{(Bayes' Theorem)}\label{e:bayes} \\ P(\bm{X}\lvert I) &= \int\textrm{d}\bm{Y} P(\bm{X}, \bm{Y}\lvert I) =\int\textrm{d}\bm{Y} P(\bm{X}\lvert\bm{Y}, I)\times P(\bm{Y}\lvert I) &\textrm{(Marginalization)} \label{e:marginalization} \end{align*}$$
$$\begin{align*} P(\textrm{Hypothesis}\lvert \textrm{Data}, I) &= \frac{P(\textrm{Data}\lvert \textrm{Hypothesis}, I)\times P(\textrm{Hypothesis}\lvert I)}{P(\textrm{Data}, I)}\label{e:bayes_hypothesis} \end{align*}$$$$\begin{align*} P(\bm{\Theta}\lvert \bm{D}, I) &= \frac{P(\bm{D}\lvert\bm{\Theta}, I) \times P(\bm{\Theta}\lvert I)}{P(\bm{D}\lvert I)} \label{e:bayes_theta}\\ &\propto P(\bm{D}\lvert\bm{\Theta}, I)\times P(\bm{\Theta}\lvert I) \end{align*}$$

$\bm{D} = \{D_k\}_{k=1}^N$ $$\begin{align*} P(\bm{\Theta}\lvert \bm{D},I) &\propto P(\{D_k\}_{k=1}^N\lvert \bm{\Theta}, I)\times P(\bm{\Theta}\lvert I) \\ &= P(\{D_k\}_{k=2}^N\lvert D_1, \bm{\Theta}, I) \times P(D_1\lvert \bm{\Theta}, I)\times P(\bm{\Theta}\lvert I) \\ &= P(\{D_k\}_{k=2}^N\lvert \bm{\Theta}, I)\times P(D_1\lvert \bm{\Theta}, I)\times P(\bm{\Theta}\lvert I) \\ &= P(\{D_k\}_{k=3}^N\lvert \bm{\Theta}, I)\times P(D_2\lvert \bm{\Theta}, I)\times P(D_1\lvert \bm{\Theta}, I) \times P(\bm{\Theta}\lvert I) \\ &\vdots \\ &= \big(\Pi_{k=1}^N P(D_k\lvert \bm{\Theta}, I)\big) \times P(\bm{\Theta}\lvert I) \end{align*}$$

$$\begin{align*} P(\bm{\Theta}\lvert \bm{D},I)&= \big(\Pi_{k=m+1}^{N} P(D_k\lvert \bm{\Theta}, I)\big) \big(\Pi_{j=1}^{m} P(D_j\lvert\bm{\Theta}, I)\big) P(\bm{\Theta}\lvert I) \\ &=\big(\Pi_{k=m+1}^{N} P(D_k\lvert \bm{\Theta}, I)\big) \times P(\bm{\Theta}\lvert \{D_j\}_{j=1}^m, I) \end{align*}$$
$$\begin{align*} L &= \ln{P(\bm{\Theta}\lvert \bm{D}, I)} = \ln P \\ &= L\lvert_{\bm{\Theta}_0} + \bm{\nabla}L\lvert_{\bm{\Theta}_0}\cdot(\bm{\Theta} - \bm{\Theta}_0) + \frac{1}{2} (\bm{\Theta}-\bm{\Theta}_0)^{t}\cdot\bm{\nabla}\bm{\nabla}L\lvert_{\bm{\Theta}_0}\cdot(\bm{\Theta}-\bm{\Theta}_0) + \mathcal{O}((\bm{\Theta}-\bm{\Theta}_0)^3) \\ &\simeq L\lvert_{\bm{\Theta}_0} + \frac{1}{2} (\bm{\Theta}-\bm{\Theta}_0)^{t}\cdot\bm{\nabla}\bm{\nabla}L\lvert_{\bm{\Theta}_0}\cdot(\bm{\Theta}-\bm{\Theta}_0) \end{align*}$$
$$\begin{align*} P(\bm{\Theta}\lvert \bm{D},I)\propto \exp{\left[-\frac{1}{2} \left(\bm{\Theta}-\bm{\Theta}_0\right)^t \cdot \left(-\bm{\nabla}\bm{\nabla}L\lvert_{\bm{\Theta}_0}\right)\cdot\left(\bm{\Theta}-\bm{\Theta}_0\right)\right]} \end{align*}$$$$\begin{align*} P_{\textrm{Gaussian}}(\bm{x}\lvert \bm{\mu}, \bm{\Sigma}) &= \frac{1}{\sqrt{(2\pi)^n \det{\bm{\Sigma}}}} \exp{\left[-\frac{1}{2}\left(\bm{x}-\bm{\mu}\right)^t \cdot\bm{\Sigma}^{-1}\cdot\left(\bm{x}-\bm{\mu}\right)\right]} \\ \left\langle\bm{x} \right\rangle &= \bm{\mu} \\ \left\langle(x_i - \mu_i)(x_j - \mu_j)\right\rangle &= \Sigma_{ij} \end{align*}$$$$\begin{align*} \bm{\mu}&\rightarrow \bm{\Theta}_0 \\ \bm{\Sigma}&\rightarrow (-\bm{\nabla}\bm{\nabla}L\lvert_{\bm{\Theta}_0})^{-1} \end{align*}$$
